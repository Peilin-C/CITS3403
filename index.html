    function doLogout() {

      document.getElementById('nav-user').textContent = '';

      document.getElementById('logout-btn').style.display = 'none';

    }


    // ── Shared helpers ──────────────────────────────────

    function showAlert(id, type, msg) {

      const el = document.getElementById(id);

      el.className = 'alert ' + type;

      el.textContent = msg;

    }

    function clearAlert(id) {

      const el = document.getElementById(id);

      el.className = 'alert';

      el.textContent = '';

    }

    function setFieldError(inputEl, msgId, msg) {

      inputEl.className = 'error';

      const m = document.getElementById(msgId);

      m.className = 'field-msg error';

      m.textContent = '⚠ ' + msg;

    }

    function clearFieldError(inputEl) {

      inputEl.className = '';

      const msgEl = inputEl.nextElementSibling;

      if (msgEl && msgEl.classList.contains('field-msg')) {

        msgEl.className = 'field-msg';

        msgEl.textContent = '';

      }

    }
